  <script>
    // Inisialisasi kamera dan AR
    document.addEventListener('DOMContentLoaded', () => {
        const btnCamera = document.getElementById('btnCamera');
        const arSection = document.getElementById('ar-section');
        const video = document.getElementById('camera-video');
        const modelViewer = document.querySelector('model-viewer');

    // Fungsi untuk update label cuaca
    const updateWeatherLabel = () => {
      const weatherLabel = document.querySelector('.weather-label');
      const suhu = document.getElementById('suhu')?.textContent || '';
      const kondisi = document.getElementById('kondisi')?.textContent || '';
      weatherLabel.textContent = `${kondisi} ${suhu} üå§Ô∏è`;
    };

    // Debug info
    console.log('Initializing viewer...');
    if (customElements.get('model-viewer')) {
      console.log('model-viewer is supported and loaded');
    } else {
      console.error('model-viewer is not loaded properly');
    }

    btnCamera.addEventListener('click', async () => {
      // Tampilkan section AR
      arSection.classList.remove('hidden');

      let stream = null;
      try {
        // Minta izin akses kamera (coba untuk semua platform)
        stream = await navigator.mediaDevices.getUserMedia({
          video: { facingMode: 'environment', width: { ideal: 1280 }, height: { ideal: 720 } },
          audio: false
        });

        // Set video stream dan play
        video.srcObject = stream;
        await video.play();

        // Tampilkan model 3D (small overlay) setelah kamera aktif
        const mv = document.getElementById('mv');
        const weatherLabel = document.querySelector('.weather-label');
        if (mv) {
          mv.style.display = 'block';
          // Ensure camera-controls enabled so user can rotate 360¬∞ manually
          mv.setAttribute('camera-controls', '');
          // Make sure auto-rotate is disabled so user controls rotation
          mv.removeAttribute('auto-rotate');
          mv.addEventListener('load', () => console.log('3D model loaded'));
        }

        // Show weather label and update it immediately and periodically
        if (weatherLabel) weatherLabel.style.display = 'block';
        updateWeatherLabel();
        setInterval(updateWeatherLabel, 3000);

      } catch (error) {
        // Informative error handling in Indonesian
        let errorMessage = 'Terjadi kesalahan saat mengakses kamera:\n\n';
        if (error && error.name === 'NotAllowedError') {
          errorMessage += 'Akses kamera ditolak. Mohon izinkan akses kamera di browser Anda.';
        } else if (error && error.name === 'NotFoundError') {
          errorMessage += 'Tidak dapat menemukan kamera. Pastikan perangkat Anda memiliki kamera.';
        } else if (error && error.name === 'NotReadableError') {
          errorMessage += 'Kamera sedang digunakan oleh aplikasi lain. Tutup aplikasi tersebut dan coba lagi.';
        } else {
          errorMessage += error?.message || String(error);
        }
        alert(errorMessage);
        console.error('Error akses kamera:', error);

  // Jika gagal, sembunyikan video, model, dan label
  if (video) video.style.display = 'none';
  const mvErr = document.getElementById('mv');
  const labelErr = document.querySelector('.weather-label');
  if (mvErr) mvErr.style.display = 'none';
  if (labelErr) labelErr.style.display = 'none';
      }
    });
    });
  </script>
